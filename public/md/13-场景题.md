### 16.3 JWT vs Session

```javascript
// JWT（JSON Web Token）
// 优点：无状态、适合分布式、跨域方便
// 缺点：无法主动失效、Token 体积大、安全性依赖客户端

const jwt = require('jsonwebtoken');

function generateToken(payload) {
  return jwt.sign(payload, process.env.JWT_SECRET, { expiresIn: '7d' });
}

function verifyToken(token) {
  return jwt.verify(token, process.env.JWT_SECRET);
}

// JWT 刷新机制
app.post('/api/refresh', (req, res) => {
  const { refreshToken } = req.body;
  const decoded = verifyToken(refreshToken);
  const newAccessToken = generateToken({ userId: decoded.userId });
  res.json({ token: newAccessToken });
});

// Session
// 优点：服务端控制、安全性高、Token 体积小
// 缺点：需要服务端存储、不适合分布式（需要共享 Session）

const session = require('express-session');
```

```javascript
// RBAC 中间件
function checkPermission(requiredPermissions) {
  return (req, res, next) => {
    const user = req.user;
    const userPermissions = user.role.permissions.map(p => p.code);

    const hasPermission = requiredPermissions.every(
      perm => userPermissions.includes(perm)
    );

    if (!hasPermission) {
      return res.status(403).json({ error: '权限不足' });
    }
    next();
  };
}
```

## 18. 单点登录 SSO

### 18.1 原理

```
SSO（Single Sign-On）核心思想：
  用户在认证中心登录一次，即可访问所有相互信任的应用系统。

CAS（Central Authentication Service）流程：
  1. 用户访问应用A → 未登录 → 重定向到认证中心
  2. 认证中心检测到用户未登录 → 显示登录页
  3. 用户在认证中心登录 → 认证中心创建全局 Session
  4. 认证中心重定向回应用A，携带 Ticket
  5. 应用A用 Ticket 向认证中心验证 → 获取用户信息 → 创建本地 Session
  6. 用户访问应用B → 重定向到认证中心 → 已登录 → 直接返回 Ticket
  7. 应用B用 Ticket 验证 → 获取用户信息 → 创建本地 Session
```

### 18.2 实现方案

```javascript
// 基于 Cookie + Redis 的简单 SSO
// 前提：所有子域共享认证 Cookie

// 认证中心（auth.example.com）
app.post('/api/login', async (req, res) => {
  const { username, password } = req.body;
  const user = await verifyUser(username, password);

  // 生成 SSO Token，存入 Redis
  const ssoToken = generateToken();
  await redis.setEx(`sso:${ssoToken}`, 86400, JSON.stringify(user));

  // 设置跨域 Cookie（所有子域共享）
  res.cookie('sso_token', ssoToken, {
    domain: '.example.com',
    path: '/',
    httpOnly: true,
    secure: true,
    maxAge: 86400000
  });

  // 重定向回子应用
  const redirectUrl = req.query.redirect;
  res.redirect(`${redirectUrl}?sso_token=${ssoToken}`);
});

// 子应用验证 Token
app.get('/api/sso/verify', async (req, res) => {
  const { sso_token } = req.query;
  const userData = await redis.get(`sso:${sso_token}`);
  if (!userData) return res.status(401).json({ error: 'Token 无效' });

  const user = JSON.parse(userData);
  // 创建子应用本地 Session
  req.session.user = user;
  res.json({ success: true, user });
});
```

```javascript
class Tracker {
  constructor(config) {
    this.appId = config.appId;
    this.serverUrl = config.serverUrl;
    this.queue = [];
    this.maxQueueSize = 20;
    this.flushInterval = 5000;
    this.userId = this.getUserId();
    this.sessionId = this.getSessionId();

    // 自动采集
    this.trackPageView();
    this.trackError();
    this.trackPerformance();
  }

  // 事件上报
  track(eventName, data = {}) {
    const event = {
      eventId: generateId(),
      eventName,
      data,
      userId: this.userId,
      sessionId: this.sessionId,
      timestamp: Date.now(),
      url: window.location.href,
      userAgent: navigator.userAgent
    };

    this.queue.push(event);
    if (this.queue.length >= this.maxQueueSize) {
      this.flush();
    }
  }

  // 批量上报
  flush() {
    if (this.queue.length === 0) return;
    const events = [...this.queue];
    this.queue = [];

    navigator.sendBeacon(this.serverUrl, JSON.stringify(events));
    // 降级：fetch with keepalive
  }

  // PV 采集
  trackPageView() {
    this.track('page_view', {
      title: document.title,
      referrer: document.referrer
    });
  }

  // 错误采集
  trackError() {
    window.addEventListener('error', (e) => {
      this.track('js_error', {
        message: e.message,
        filename: e.filename,
        lineno: e.lineno,
        colno: e.colno,
        stack: e.error?.stack
      });
    });

    window.addEventListener('unhandledrejection', (e) => {
      this.track('promise_error', {
        reason: e.reason
      });
    });
  }

  // 性能采集
  trackPerformance() {
    window.addEventListener('load', () => {
      const timing = performance.getEntriesByType('navigation')[0];
      this.track('performance', {
        dns: timing.domainLookupEnd - timing.domainLookupStart,
        tcp: timing.connectEnd - timing.connectStart,
        ttfb: timing.responseStart - timing.requestStart,
        domReady: timing.domContentLoadedEventEnd - timing.fetchStart,
        load: timing.loadEventEnd - timing.fetchStart
      });
    });
  }
}

// 使用
const tracker = new Tracker({
  appId: 'my-app',
  serverUrl: 'https://analytics.example.com/collect'
});

// 手动埋点
tracker.track('button_click', { buttonId: 'submit', page: 'checkout' });
```

```javascript
  });

  }
```

## 21. 倒计时/秒杀场景

### 21.1 前端倒计时

```javascript
// 秒杀倒计时
class Countdown {
  constructor(endTime, onTick, onEnd) {
    this.endTime = endTime;
    this.onTick = onTick;
    this.onEnd = onEnd;
    this.timer = null;
    this.start();
  }

  start() {
    this.tick();
    this.timer = setInterval(() => this.tick(), 1000);
  }

  tick() {
    const now = Date.now();
    const remaining = this.endTime - now;

    if (remaining <= 0) {
      clearInterval(this.timer);
      this.onTick({ days: 0, hours: 0, minutes: 0, seconds: 0 });
      this.onEnd();
      return;
    }

    const days = Math.floor(remaining / (1000 * 60 * 60 * 24));
    const hours = Math.floor((remaining % (1000 * 60 * 60 * 24)) / (1000 * 60 * 60));
    const minutes = Math.floor((remaining % (1000 * 60 * 60)) / (1000 * 60));
    const seconds = Math.floor((remaining % (1000 * 60)) / 1000);

    this.onTick({ days, hours, minutes, seconds });
  }

  stop() {
    clearInterval(this.timer);
  }
}

// 使用
const countdown = new Countdown(
  Date.now() + 3600000, // 1小时后
  (time) => {
    document.getElementById('timer').textContent =
      `${time.hours}:${String(time.minutes).padStart(2, '0')}:${String(time.seconds).padStart(2, '0')}`;
  },
  () => {
    console.log('倒计时结束！');
    // 启用抢购按钮
  }
);
```
