import hashlib
import base62

def generate_short_code(long_url: str, salt: str = "") -> str:
    """生成短码（方案C：哈希 + 冲突检测）"""
    hash_input = f"{long_url}{salt}"
    hash_val = hashlib.md5(hash_input.encode()).hexdigest()[:8]
    short_code = base62.encode(int(hash_val, 16))[:7]
    return short_code

# Saga模式：订单创建流程
# 步骤：创建订单 -> 扣减库存 -> 创建支付 -> 发送通知
# 每步有对应的补偿操作

class CreateOrderSaga:
    def execute(self, order_data):
        try:
            # 步骤1：创建订单
            order = self.create_order(order_data)
            # 步骤2：扣减库存
            self.deduct_inventory(order.items)
            # 步骤3：创建支付单
            self.create_payment(order)
            # 步骤4：发送通知
            self.send_notification(order)
            return {"status": "success", "order": order}
        except Exception as e:
            # 补偿：按逆序回滚
            self.compensate(order)
            raise

    def compensate(self, order):
        """补偿操作（逆序执行）"""
        if order.payment_created:
            self.cancel_payment(order)
        if order.inventory_deducted:
            self.restore_inventory(order.items)
        if order.created:
            self.cancel_order(order.id)

class ConnectionManager:

    def __init__(self):
        self.active_connections: Dict[int, WebSocket] = {}

    async def connect(self, user_id: int, websocket: WebSocket):
        await websocket.accept()
        self.active_connections[user_id] = websocket

        self.active_connections.pop(user_id, None)
        redis.srem("online_users", user_id)

    async def send_personal(self, user_id: int, message: dict):
        ws = self.active_connections.get(user_id)
        if ws:
            await ws.send_json(message)

    async def broadcast(self, message: dict):
        for ws in self.active_connections.values():
            await ws.send_json(message)

### 1. 限流算法详解（⭐高频）

#### 固定窗口（计数器）

```python
import time

class FixedWindowRateLimiter:
    """固定窗口限流：每个固定时间窗口内最多允许 N 次请求"""
    def __init__(self, limit, window_seconds):
        self.limit = limit
        self.window_seconds = window_seconds
        self.count = 0
        self.window_start = time.time()

    def allow(self):
        now = time.time()
        if now - self.window_start >= self.window_seconds:
            # 窗口重置
            self.count = 0
            self.window_start = now

        if self.count < self.limit:
            self.count += 1
            return True
        return False

# 问题：临界点突发（窗口边界处可能 2 倍流量）
# 例如：limit=100, window=1s
# 0.99s 时来了100个请求，1.01s 又来了100个请求
# 0.02s 内实际通过了200个请求
```

#### 滑动窗口（Redis ZSet 实现）

```python
import redis
import time

class SlidingWindowRateLimiter:
    """滑动窗口限流：基于 Redis ZSet"""
    def __init__(self, redis_client, key, limit, window_seconds):
        self.redis = redis_client
        self.key = key
        self.limit = limit
        self.window_seconds = window_seconds

    def allow(self):
        now = time.time()
        window_start = now - self.window_seconds

        # 1. 移除窗口外的旧记录
        self.redis.zremrangebyscore(self.key, 0, window_start)

        # 2. 统计当前窗口内的请求数
        count = self.redis.zcard(self.key)

        if count < self.limit:
            # 3. 添加新记录（用时间戳作为 score 和 member）
            self.redis.zadd(self.key, {str(now): now})
            # 4. 设置过期时间（防止 key 永久存在）
            self.redis.expire(self.key, self.window_seconds)
            return True
        return False

# 使用
r = redis.Redis()
limiter = SlidingWindowRateLimiter(r, "rate_limit:user:123", limit=100, window_seconds=60)
if limiter.allow():
    print("请求通过")
else:
    print("请求被限流")
```

#### 漏桶（恒定速率出）

```python
import time
import threading

class LeakyBucket:
    """漏桶限流：恒定速率处理请求，超出的请求排队或丢弃"""
    def __init__(self, rate, capacity):
        self.rate = rate          # 每秒处理多少个请求
        self.capacity = capacity  # 桶的容量（最多排队多少个）
        self.queue = []
        self.last_leak = time.time()
        self.lock = threading.Lock()

    def allow(self):
        with self.lock:
            now = time.time()
            # 漏出请求
            elapsed = now - self.last_leak
            leak_count = int(elapsed * self.rate)
            if leak_count > 0:
                self.queue = self.queue[leak_count:]
                self.last_leak = now

            if len(self.queue) < self.capacity:
                self.queue.append(1)
                return True
            return False

# 特点：匀速处理，不能应对突发流量
# 适用场景：保护下游服务，确保请求速率恒定
```

#### 令牌桶（允许突发）

```python
import time
import threading

class TokenBucket:
    """令牌桶限流：允许突发流量，但平均速率受限"""
    def __init__(self, rate, capacity):
        self.rate = rate          # 每秒生成多少个令牌
        self.capacity = capacity  # 桶的最大容量
        self.tokens = capacity    # 初始满桶
        self.last_refill = time.time()
        self.lock = threading.Lock()

    def allow(self):
        with self.lock:
            now = time.time()
            # 补充令牌
            elapsed = now - self.last_refill
            self.tokens = min(self.capacity, self.tokens + elapsed * self.rate)
            self.last_refill = now

            if self.tokens >= 1:
                self.tokens -= 1
                return True
            return False

# 特点：允许突发（桶内有令牌时），但平均速率 = rate
# 适用场景：API 限流（如 100次/分钟，允许短时突发）
# Google Guava RateLimiter 就是令牌桶实现
```

#### 雪花算法（Snowflake）原理与实现

```
64位 ID 结构：
0 | 41位时间戳 | 10位机器ID | 12位序列号

- 1位符号位：固定为0
- 41位时间戳：毫秒级，可用约69年
- 10位机器ID：1024台机器（可拆分为数据中心ID + 机器ID）
- 12位序列号：每毫秒最多4096个ID
```

```python
import time
import threading

class Snowflake:
    def __init__(self, datacenter_id=0, worker_id=0):
        # 起始时间戳（2024-01-01）
        self.twepoch = 1704067200000

        self.datacenter_id_bits = 5
        self.worker_id_bits = 5
        self.sequence_bits = 12

        self.max_datacenter_id = -1 ^ (-1 << self.datacenter_id_bits)  # 31
        self.max_worker_id = -1 ^ (-1 << self.worker_id_bits)          # 31
        self.max_sequence = -1 ^ (-1 << self.sequence_bits)            # 4095

        self.worker_id_shift = self.sequence_bits                      # 12
        self.datacenter_id_shift = self.sequence_bits + self.worker_id_bits  # 17
        self.timestamp_shift = self.sequence_bits + self.worker_id_bits + self.datacenter_id_bits  # 22

        if datacenter_id > self.max_datacenter_id or datacenter_id < 0:
            raise ValueError("datacenter_id 超出范围")
        if worker_id > self.max_worker_id or worker_id < 0:
            raise ValueError("worker_id 超出范围")

        self.datacenter_id = datacenter_id
        self.worker_id = worker_id
        self.sequence = 0
        self.last_timestamp = -1
        self.lock = threading.Lock()

    def _current_millis(self):
        return int(time.time() * 1000)

    def _wait_next_millis(self, last_timestamp):
        timestamp = self._current_millis()
        while timestamp <= last_timestamp:
            timestamp = self._current_millis()
        return timestamp

    def next_id(self):
        with self.lock:
            timestamp = self._current_millis()

            if timestamp < self.last_timestamp:
                raise RuntimeError("时钟回拨，拒绝生成ID")

            if timestamp == self.last_timestamp:
                self.sequence = (self.sequence + 1) & self.max_sequence
                if self.sequence == 0:
                    timestamp = self._wait_next_millis(self.last_timestamp)
            else:
                self.sequence = 0

            self.last_timestamp = timestamp

            new_id = ((timestamp - self.twepoch) << self.timestamp_shift) | \
                     (self.datacenter_id << self.datacenter_id_shift) | \
                     (self.worker_id << self.worker_id_shift) | \
                     self.sequence
            return new_id

# 使用
snowflake = Snowflake(datacenter_id=1, worker_id=1)
for _ in range(5):
    print(snowflake.next_id())
```

#### Leaf-segment（号段模式）

### 3. 分布式锁对比（⭐高频）

#### Redis SETNX（AP）

```python
import redis
import uuid
import time

class RedisDistributedLock:
    """基于 Redis SETNX 的分布式锁"""
    def __init__(self, redis_client, lock_name, expire_seconds=10):
        self.redis = redis_client
        self.lock_name = f"lock:{lock_name}"
        self.expire_seconds = expire_seconds
        self.identifier = str(uuid.uuid4())  # 唯一标识，防止误删

    def acquire(self, timeout=10):
        """获取锁（带超时重试）"""
        end_time = time.time() + timeout
        while time.time() < end_time:
            # SET key value NX EX seconds（原子操作）
            if self.redis.set(self.lock_name, self.identifier, nx=True, ex=self.expire_seconds):
                return True
            time.sleep(0.01)  # 短暂等待
        return False

    def release(self):
        """释放锁（Lua 脚本保证原子性）"""
        lua_script = """
        if redis.call("get", KEYS[1]) == ARGV[1] then
            return redis.call("del", KEYS[1])
        else
            return 0
        end
        """
        self.redis.eval(lua_script, 1, self.lock_name, self.identifier)

# 使用
r = redis.Redis()
lock = RedisDistributedLock(r, "order:1001")
if lock.acquire(timeout=5):
    try:
        # 执行业务逻辑
        print("获取锁成功，执行业务...")
    finally:
        lock.release()
```

### 4. 一致性哈希（⭐高频）

#### 哈希环原理

```
一致性哈希将整个哈希空间组织成一个虚拟的环（0 ~ 2^32 - 1）。

1. 将节点（服务器）的 IP/名称哈希到环上
2. 将数据的 key 哈希到环上
3. 顺时针查找，数据存储在最近的节点上

        0
    N3 ---- N1
    |        |
    |        |
    N2 ---- N4
     2^32-1

例如：key1 哈希到 N3 和 N1 之间，存储在 N1 上
```

#### 虚拟节点解决倾斜

```
问题：节点少时，哈希分布不均匀，某些节点负载过高

解决：每个物理节点对应多个虚拟节点

物理节点 N1 --> 虚拟节点 N1#1, N1#2, N1#3, N1#4
物理节点 N2 --> 虚拟节点 N2#1, N2#2, N2#3, N2#4

虚拟节点越多，分布越均匀（通常 150~200 个虚拟节点/物理节点）
```

#### 数据迁移最小化

```
传统哈希取模：N 台机器变为 N+1 台，几乎所有数据都要迁移
一致性哈希：新增/删除一个节点，只影响环上相邻节点的数据

例如：3节点 -> 4节点
- 哈希取模：约 75% 数据需要迁移
- 一致性哈希：约 25% 数据需要迁移
```

#### 与哈希取模的对比

| 特性 | 哈希取模 | 一致性哈希 |
|------|---------|-----------|
| 数据分布 | 均匀 | 取决于虚拟节点数 |
| 扩容影响 | 大量数据迁移 | 仅相邻节点数据迁移 |
| 缩容影响 | 大量数据迁移 | 仅相邻节点数据迁移 |
| 实现复杂度 | 简单 | 中等 |
| 适用场景 | 节点数固定 | 节点动态增减 |

#### 手写实现

```python
import hashlib
import bisect

class ConsistentHashing:
    """一致性哈希实现"""
    def __init__(self, virtual_nodes=150):
        self.virtual_nodes = virtual_nodes
        self.ring = {}       # {hash_value: node_name}
        self.sorted_keys = []  # 排序后的哈希值列表

    def _hash(self, key):
        """使用 MD5 哈希"""
        return int(hashlib.md5(key.encode()).hexdigest(), 16)

    def add_node(self, node):
        """添加节点（含虚拟节点）"""
        for i in range(self.virtual_nodes):
            virtual_key = f"{node}#vn{i}"
            hash_val = self._hash(virtual_key)
            if hash_val not in self.ring:
                self.ring[hash_val] = node
                bisect.insort(self.sorted_keys, hash_val)

    def remove_node(self, node):
        """移除节点"""
        for i in range(self.virtual_nodes):
            virtual_key = f"{node}#vn{i}"
            hash_val = self._hash(virtual_key)
            if hash_val in self.ring:
                del self.ring[hash_val]
                self.sorted_keys.remove(hash_val)

    def get_node(self, key):
        """获取 key 应该存储的节点"""
        if not self.sorted_keys:
            return None
        hash_val = self._hash(key)
        # 顺时针查找第一个节点
        idx = bisect.bisect_left(self.sorted_keys, hash_val)
        if idx >= len(self.sorted_keys):
            idx = 0  # 环形
        return self.ring[self.sorted_keys[idx]]

# 使用
ch = ConsistentHashing(virtual_nodes=100)
ch.add_node("node1")
ch.add_node("node2")
ch.add_node("node3")

print(ch.get_node("user:1001"))  # node2
print(ch.get_node("user:1002"))  # node1
print(ch.get_node("user:1003"))  # node3

# 新增节点后，只有部分 key 的映射会变化
ch.add_node("node4")
print(ch.get_node("user:1001"))  # 可能还是 node2，也可能变了
```

### 5. CAP 定理与 BASE 理论（⭐高频）

#### CAP 定理

```
CAP 定理：分布式系统不可能同时满足 C、A、P 三者，最多只能同时满足两个。

C (Consistency 一致性)：所有节点在同一时刻看到的数据一致
A (Availability 可用性)：每个请求都能在合理时间内收到非错误响应
P (Partition tolerance 分区容错)：网络分区发生时，系统仍能继续工作

由于网络分区（P）在分布式系统中不可避免，实际选择只有：
- CP：保证一致性，牺牲可用性（如 ZooKeeper、etcd）
- AP：保证可用性，牺牲一致性（如 Cassandra、Eureka）
```

| 组合 | 含义 | 代表系统 | 适用场景 |
|------|------|---------|---------|
| CP | 一致性 + 分区容错 | ZooKeeper, etcd, HBase | 配置中心、元数据管理 |
| AP | 可用性 + 分区容错 | Cassandra, Eureka, DynamoDB | 缓存、社交Feed |
| CA | 一致性 + 可用性 | 传统单机数据库 | 非分布式系统 |

#### BASE 理论

```
BASE 是 CAP 中 AP 方案的延伸：

BA (Basically Available) 基本可用：允许损失部分可用性（如响应时间变长）
S (Soft State) 软状态：允许数据存在中间状态
E (Eventually Consistent) 最终一致性：数据最终会达到一致状态
```

#### 实际案例分析

```
案例1：微博热搜（AP + 最终一致性）
- 发微博后，粉丝可能延迟几秒看到 --> 基本可用
- 热搜排名可能各节点短暂不一致 --> 软状态
- 最终所有节点数据一致 --> 最终一致性

案例2：银行转账（CP + 强一致性）
- 转账必须保证双方余额同时更新
- 宁可暂时不可用，也不能数据不一致
- 使用分布式事务（2PC / TCC / Saga）

案例3：电商库存（AP + 最终一致性）
- Redis 预扣库存（允许短暂超卖）
- MQ 异步扣减数据库库存
- 最终通过补偿机制保证一致
```

#### 面试问答

**Q: 你在项目中如何处理一致性和可用性的平衡？**
答：根据业务场景选择。订单创建用 CP（强一致性），保证不超卖；用户Feed流用 AP（最终一致性），保证高可用和低延迟。核心原则是：不是所有数据都需要强一致性，大部分场景最终一致性就够了。

#### 代理模式

```python
import time
from functools import wraps

def timing_proxy(cls):
    """代理模式：为类的所有方法添加计时功能"""
    for attr_name, attr_value in cls.__dict__.items():
        if callable(attr_value):
            def make_wrapper(original_func):
                @wraps(original_func)
                def wrapper(*args, **kwargs):
                    start = time.time()
                    result = original_func(*args, **kwargs)
                    print(f"{original_func.__name__} 耗时: {time.time() - start:.3f}s")
                    return result
                return wrapper
            setattr(cls, attr_name, make_wrapper(attr_value))
    return cls

@timing_proxy
class UserService:
    def get_user(self, user_id):
        time.sleep(0.5)
        return {"id": user_id, "name": "Alice"}
```

#### 设计一个 Feed 流/Timeline 系统

```
需求：
- 支持百万级用户
- 每个用户有关注列表
- 需要支持分页、排序
- 延迟 < 200ms

方案选择：

1. 推模式（Write-heavy）：发微博时推送到所有粉丝的收件箱
   - 写放大：100万粉丝的明星发一条微博 = 100万次写入
   - 适合：粉丝数少的普通用户

2. 拉模式（Read-heavy）：读Feed时，拉取关注用户的最新微博
   - 读放大：关注了1000人 = 1000次读取 + 排序合并
   - 适合：粉丝数多的明星

3. 推拉结合（推荐方案）：
   - 大V（粉丝 > 阈值）用拉模式
   - 普通用户用推模式
   - 粉丝收件箱 = 推的普通用户微博 + 拉的大V微博

存储设计：
- 微博内容：MySQL（持久化）+ Redis（缓存）
- 用户收件箱：Redis ZSet（score = 时间戳）
- 关注关系：Redis Set / MySQL

读Feed流程：
1. 先从 Redis ZSet 读取收件箱（推模式部分）
2. 再拉取大V的最新微博（拉模式部分）
3. 合并排序，分页返回
4. 缓存结果（Redis，5分钟过期）
```
